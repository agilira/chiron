Step 1: Aggiungi Dati al Frontmatter

Step 2: Usa il Shortcode con `data` Attribute

Step 3: Chiron Genera il Data Island

Durante il build, Chiron genera:

<!-- Data Island - injected at build time -->
<script type="application/json" id="related-posts-widget-data">
{
  "title": "My Blog Post",
  "tags": ["react", "performance", "seo"],
  "relatedPosts": [
    { "title": "First Related", "url": "/posts/first" },
    { "title": "Second Related", "url": "/posts/second" }
  ]
}
</script>

<!-- Lazy-loaded container -->
<div class="lazy-app-container" 
     data-lazy-app="vanilla"
     data-script-src="/assets/widgets/related-posts.js"
     id="related-posts-widget">
  <div class="placeholder">Loading related posts...</div>
</div>

Step 4: L'App Legge i Dati

// assets/widgets/related-posts.js
(function() {
  'use strict';
  
  document.addEventListener('lazy-app-loaded', function(event) {
    if (event.detail.appId !== 'related-posts-widget') return;
    
    const container = event.detail.container;
    
    // Read injected Data Island
    const dataScript = document.getElementById('related-posts-widget-data');
    const data = JSON.parse(dataScript.textContent);
    
    // Render with zero HTTP requests!
    const html = `
      <div class="related-posts">
        <h3>Related to: ${data.title}</h3>
        <ul>
          ${data.relatedPosts.map(post => `
            <li><a href="${post.url}">${post.title}</a></li>
          `).join('')}
        </ul>
        <div class="tags">
          ${data.tags.map(tag => `<span class="tag">${tag}</span>`).join('')}
        </div>
      </div>
    `;
    
    container.innerHTML = html;
  });
})();

üîß Sintassi Completa

Attributo `data`

data="key1,key2.nested,key3.deep.value"
  • Separator: virgola (,)
  • Dot Notation: naviga oggetti nested
  • Property Name: ultimo elemento del path diventa chiave nel JSON

Esempi

# Semplice
data="page.title"
‚Üí { "title": "Page Title" }

# Multiple keys
data="page.title,page.author"
‚Üí { "title": "...", "author": "..." }

# Nested objects
data="page.customData.settings.theme"
‚Üí { "theme": "dark" }

# Arrays
data="page.tags,page.categories"
‚Üí { "tags": [...], "categories": [...] }

# Complex objects
data="page.metadata,config.project"
‚Üí { "metadata": {...}, "project": {...} }
